> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a safe `ireality` CLI that evaluates explicit candidate domains for Xray REALITY target-domain compatibility and ranks recommendations.

**Architecture:** Implement a small Python package with separate modules for models, candidate loading, TLS probing, HTTP probing, scoring, and CLI rendering. Keep network probing separate from scoring so most tests are deterministic and do not require live network access.

**Tech Stack:** Python 3.11+, Typer, Rich, pytest, standard-library `ssl`, `socket`, `http.client`, `json`, and dataclasses.

from __future__ import annotations

from dataclasses import asdict, dataclass
from statistics import mean, pstdev


@dataclass(frozen=True)
class TlsProbeResult:
    domain: str
    tcp_connected: bool
    tls_handshake: bool
    certificate_valid: bool
    hostname_matches: bool
    alpn_protocol: str | None
    latency_ms: list[float]
    subject: str | None
    issuer: str | None
    san: list[str]
    not_before: str | None
    not_after: str | None
    error_code: str | None
    error_message: str | None

    @property
    def average_latency_ms(self) -> float | None:
        if not self.latency_ms:
            return None
        return mean(self.latency_ms)

    @property
    def jitter_ms(self) -> float | None:
        if len(self.latency_ms) < 2:
            return None
        return pstdev(self.latency_ms)

    def to_dict(self) -> dict[str, object]:
        data = asdict(self)
        data["average_latency_ms"] = self.average_latency_ms
        data["jitter_ms"] = self.jitter_ms
        return data


@dataclass(frozen=True)
class HttpProbeResult:
    domain: str
    status_code: int | None
    server: str | None
    error_code: str | None
    error_message: str | None

    def to_dict(self) -> dict[str, object]:
        return asdict(self)


@dataclass(frozen=True)
class ScoreResult:
    domain: str
    score: int
    verdict: str
    positive_reasons: list[str]
    negative_reasons: list[str]

    def to_dict(self) -> dict[str, object]:
        return asdict(self)

from __future__ import annotations

from ireality.models import HttpProbeResult, ScoreResult, TlsProbeResult


def score_probe_result(tls: TlsProbeResult, http: HttpProbeResult | None = None) -> ScoreResult:
    score = 0
    positive: list[str] = []
    negative: list[str] = []

    if not tls.tcp_connected:
        return ScoreResult(tls.domain, 0, "reject", positive, ["TCP connection failed"])
    score += 10
    positive.append("TCP connection succeeded")

    if not tls.tls_handshake:
        return ScoreResult(tls.domain, 10, "reject", positive, ["TLS handshake failed"])
    score += 30
    positive.append("TLS handshake succeeded")

    if tls.certificate_valid:
        score += 20
        positive.append("Certificate is currently valid")
    else:
        score -= 20
        negative.append("Certificate is not currently valid")

    if tls.hostname_matches:
        score += 20
        positive.append("SNI hostname matches certificate")
    else:
        score -= 30
        negative.append("SNI hostname does not match certificate")

    if tls.alpn_protocol == "h2":
        score += 10
        positive.append("ALPN negotiated h2")
    elif tls.alpn_protocol == "http/1.1":
        score += 5
        positive.append("ALPN negotiated http/1.1")
    else:
        negative.append("No preferred ALPN protocol negotiated")

    average_latency = tls.average_latency_ms
    if average_latency is not None:
        if average_latency <= 100:
            score += 10
            positive.append("Average latency is low")
        elif average_latency <= 250:
            score += 5
            positive.append("Average latency is acceptable")
        else:
            score -= 10
            negative.append("Average latency is high")

    jitter = tls.jitter_ms
    if jitter is not None:
        if jitter <= 25:
            score += 5
            positive.append("Latency jitter is low")
        elif jitter > 100:
            score -= 5
            negative.append("Latency jitter is high")

    if http is not None:
        if http.status_code is not None and http.status_code < 500:
            score += 5
            positive.append("HTTP HEAD returned a sane response")
        elif http.error_code:
            negative.append("HTTP HEAD probe failed")

    score = max(0, min(100, score))
    if score >= 85:
        verdict = "excellent"
    elif score >= 70:
        verdict = "good"
    elif score >= 50:
        verdict = "weak"
    else:
        verdict = "reject"

    if not tls.certificate_valid or not tls.hostname_matches:
        verdict = "reject" if score < 70 else "weak"

    return ScoreResult(tls.domain, score, verdict, positive, negative)
```

from __future__ import annotations

import re
from pathlib import Path
from urllib.parse import urlparse

BUILT_IN_CANDIDATES = [
    "www.cloudflare.com",
    "www.microsoft.com",
    "www.apple.com",
    "www.amazon.com",
    "www.google.com",
    "www.github.com",
    "www.wikipedia.org",
    "www.bing.com",
]

_HOSTNAME_RE = re.compile(
    r"^(?=.{1,253}$)(?!-)(?:[a-z0-9](?:[a-z0-9-]{0,61}[a-z0-9])?\.)+[a-z]{2,63}$"
)


def normalize_domain(value: str) -> str | None:
    raw = value.strip().lower()
    if not raw or raw.startswith("#") or "*" in raw:
        return None

    parsed = urlparse(raw if "://" in raw else f"//{raw}")
    host = parsed.hostname
    if host is None:
        return None

    host = host.rstrip(".")
    if not _HOSTNAME_RE.match(host):
        return None
    return host


def load_candidates(path: Path | None = None, include_builtins: bool = True) -> list[str]:
    domains: list[str] = []

    if include_builtins:
        domains.extend(BUILT_IN_CANDIDATES)

    if path is not None:
        for line in path.read_text(encoding="utf-8").splitlines():
            domain = normalize_domain(line)
            if domain is not None:
                domains.append(domain)

    normalized = [domain for domain in (normalize_domain(item) for item in domains) if domain is not None]
    return sorted(set(normalized))
```

```python
from __future__ import annotations

import socket
import ssl
import time
from datetime import UTC, datetime
from email.utils import parsedate_to_datetime
from typing import Any

from ireality.models import TlsProbeResult

DEFAULT_TIMEOUT_SECONDS = 5.0
DEFAULT_SAMPLES = 3


def probe_tls(domain: str, port: int = 443, timeout: float = DEFAULT_TIMEOUT_SECONDS, samples: int = DEFAULT_SAMPLES) -> TlsProbeResult:
    context = ssl.create_default_context()
    context.set_alpn_protocols(["h2", "http/1.1"])

    latencies: list[float] = []
    last_cert: dict[str, Any] | None = None
    negotiated_alpn: str | None = None

    for _ in range(max(1, samples)):
        started = time.perf_counter()
        try:
            with socket.create_connection((domain, port), timeout=timeout) as raw_sock:
                raw_sock.settimeout(timeout)
                with context.wrap_socket(raw_sock, server_hostname=domain) as tls_sock:
                    cert = tls_sock.getpeercert()
                    negotiated_alpn = tls_sock.selected_alpn_protocol()
                    last_cert = cert
                    latencies.append((time.perf_counter() - started) * 1000)
        except socket.timeout as exc:
            return _failed_result(domain, latencies, "timeout", str(exc), tcp_connected=False)
        except OSError as exc:
            return _failed_result(domain, latencies, "tcp_error", str(exc), tcp_connected=False)
        except ssl.CertificateError as exc:
            return _failed_result(domain, latencies, "hostname_mismatch", str(exc), tcp_connected=True)
        except ssl.SSLError as exc:
            return _failed_result(domain, latencies, "tls_error", str(exc), tcp_connected=True)

    certificate_valid = _certificate_is_time_valid(last_cert)
    san = _extract_san(last_cert)

    return TlsProbeResult(
        domain=domain,
        tcp_connected=True,
        tls_handshake=True,
        certificate_valid=certificate_valid,
        hostname_matches=True,
        alpn_protocol=negotiated_alpn,
        latency_ms=latencies,
        subject=_format_name(last_cert.get("subject", ())) if last_cert else None,
        issuer=_format_name(last_cert.get("issuer", ())) if last_cert else None,
        san=san,
        not_before=last_cert.get("notBefore") if last_cert else None,
        not_after=last_cert.get("notAfter") if last_cert else None,
        error_code=None,
        error_message=None,
    )


def _failed_result(domain: str, latencies: list[float], code: str, message: str, tcp_connected: bool) -> TlsProbeResult:
    return TlsProbeResult(
        domain=domain,
        tcp_connected=tcp_connected,
        tls_handshake=False,
        certificate_valid=False,
        hostname_matches=False,
        alpn_protocol=None,
        latency_ms=latencies,
        subject=None,
        issuer=None,
        san=[],
        not_before=None,
        not_after=None,
        error_code=code,
        error_message=message,
    )


def _certificate_is_time_valid(cert: dict[str, Any] | None) -> bool:
    if not cert or "notAfter" not in cert:
        return False
    now = datetime.now(UTC)
    not_after = parsedate_to_datetime(cert["notAfter"])
    not_before = parsedate_to_datetime(cert["notBefore"])
    return not_before <= now <= not_after


def _extract_san(cert: dict[str, Any] | None) -> list[str]:
    if not cert:
        return []
    return [value for key, value in cert.get("subjectAltName", []) if key == "DNS"]


def _format_name(parts: tuple[tuple[tuple[str, str], ...], ...]) -> str:
    values: list[str] = []
    for group in parts:
        for key, value in group:
            values.append(f"{key}={value}")
    return ", ".join(values)
```

from __future__ import annotations

import http.client
import ssl

from ireality.models import HttpProbeResult

DEFAULT_TIMEOUT_SECONDS = 5.0


def probe_http_head(domain: str, timeout: float = DEFAULT_TIMEOUT_SECONDS) -> HttpProbeResult:
    context = ssl.create_default_context()
    connection = http.client.HTTPSConnection(domain, 443, timeout=timeout, context=context)
    try:
        connection.request("HEAD", "/", headers={"User-Agent": "ireality/0.1"})
        response = connection.getresponse()
        response.read()
        return HttpProbeResult(
            domain=domain,
            status_code=response.status,
            server=response.getheader("Server"),
            error_code=None,
            error_message=None,
        )
    except TimeoutError as exc:
        return HttpProbeResult(domain, None, None, "timeout", str(exc))
    except OSError as exc:
        return HttpProbeResult(domain, None, None, "http_error", str(exc))
    finally:
        connection.close()
```

from __future__ import annotations

import json
from pathlib import Path
from typing import Annotated

import typer
from rich.console import Console
from rich.table import Table

from ireality.candidates import load_candidates, normalize_domain
from ireality.http_probe import probe_http_head
from ireality.models import HttpProbeResult, ScoreResult, TlsProbeResult
from ireality.scoring import score_probe_result
from ireality.tls_probe import probe_tls

app = typer.Typer(help="Safe REALITY target-domain compatibility checker.")
console = Console()


def _evaluate_domain(domain: str) -> tuple[TlsProbeResult, HttpProbeResult | None, ScoreResult]:
    tls = probe_tls(domain)
    http = probe_http_head(domain) if tls.tls_handshake else None
    score = score_probe_result(tls, http)
    return tls, http, score


@app.command()
def check(
    domain: Annotated[str, typer.Argument(help="Domain to evaluate.")],
    json_output: Annotated[bool, typer.Option("--json", help="Print machine-readable JSON.")] = False,
) -> None:
    normalized = normalize_domain(domain)
    if normalized is None:
        raise typer.BadParameter("Provide a valid public domain name.")

    tls, http, score = _evaluate_domain(normalized)
    _render_single(tls, http, score, json_output)


@app.command()
def explain(
    domain: Annotated[str, typer.Argument(help="Domain to explain.")],
    json_output: Annotated[bool, typer.Option("--json", help="Print machine-readable JSON.")] = False,
) -> None:
    check(domain, json_output)


@app.command()
def scan(
    input: Annotated[Path, typer.Option("--input", "-i", exists=True, readable=True, help="File containing candidate domains.")],
    include_builtins: Annotated[bool, typer.Option(help="Include curated built-in candidates.")] = False,
    json_output: Annotated[bool, typer.Option("--json", help="Print machine-readable JSON.")] = False,
) -> None:
    domains = load_candidates(input, include_builtins=include_builtins)
    results = [_evaluate_domain(domain) for domain in domains]
    _render_many(results, json_output)


@app.command()
def recommend(
    server: Annotated[str, typer.Option("--server", help="Server hostname or IP used as report metadata.")],
    input: Annotated[Path | None, typer.Option("--input", "-i", exists=True, readable=True, help="Optional file containing extra candidates.")] = None,
    json_output: Annotated[bool, typer.Option("--json", help="Print machine-readable JSON.")] = False,
) -> None:
    domains = load_candidates(input, include_builtins=True)
    results = [_evaluate_domain(domain) for domain in domains]
    results.sort(key=lambda item: (-item[2].score, item[0].average_latency_ms or 999999))
    if json_output:
        payload = {
            "server": server,
            "results": [_result_to_dict(tls, http, score) for tls, http, score in results],
        }
        console.print(json.dumps(payload, indent=2, sort_keys=True))
        return
    console.print(f"Recommendations for server metadata: {server}")
    _render_many(results, json_output=False)


def _render_single(tls: TlsProbeResult, http: HttpProbeResult | None, score: ScoreResult, json_output: bool) -> None:
    if json_output:
        console.print(json.dumps(_result_to_dict(tls, http, score), indent=2, sort_keys=True))
        return

    table = Table(title="ireality result")
    table.add_column("Domain")
    table.add_column("Verdict")
    table.add_column("Score", justify="right")

# ireality

`ireality` is a safe command-line helper for evaluating candidate domains for Xray REALITY target-domain compatibility.

It does not steal certificates, evade defenses, exploit targets, or perform high-volume internet scanning. It performs conservative TLS/SNI/HTTP compatibility checks against explicit candidate domains and a small curated built-in candidate list.

## Install

```bash
python -m pip install -e '.[dev]'
```

## Commands

Check one domain:

```bash
ireality check example.com
```

Explain a domain with JSON output:

```bash
ireality explain example.com --json
```

Scan domains from a file:

```bash
ireality scan --input domains.txt
```

Recommend from built-in candidates, using your server as report metadata:

```bash
ireality recommend --server 203.0.113.10
```

## What it checks

- TCP 443 reachability
- TLS handshake with SNI
- Certificate validity and hostname match
- ALPN negotiation for `h2` / `http/1.1`
- Lightweight HTTPS `HEAD /`
- Latency and jitter

## Verdicts

- `excellent`: strong REALITY target candidate
- `good`: usable candidate
- `weak`: technically reachable but has drawbacks
- `reject`: unsuitable or failed required TLS checks

## Safety scope

Use this tool only for domains you are allowed to test or for ordinary public websites under conservative request rates. The tool intentionally does not support arbitrary CIDR scanning in its first implementation.
```
